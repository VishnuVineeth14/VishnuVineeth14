    <button class="game-btn" onclick="gameReset()">🔄 Reset</button>

<script>
  const canvas = document.getElementById('snakeCanvas');
  const ctx = canvas.getContext('2d');
  const gridSize = 15;
  const tileCount = 20;
  
  let snake = [{x: 10, y: 10}, {x: 9, y: 10}, {x: 8, y: 10}];
  let food = {x: 15, y: 15};
  let dx = 1;
  let dy = 0;
  let score = 0;
  let gameRunning = true;
  let nextDx = dx;
  let nextDy = dy;
  
  function drawGame() {
    ctx.fillStyle = '#0d1117';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    
    ctx.fillStyle = '#00ff41';
    ctx.shadowColor = 'rgba(0, 255, 65, 0.8)';
    ctx.shadowBlur = 10;
    for (let segment of snake) {
      ctx.fillRect(segment.x * gridSize + 1, segment.y * gridSize + 1, gridSize - 2, gridSize - 2);
    }
    ctx.shadowColor = 'transparent';
    
    ctx.fillStyle = '#ff006e';
    ctx.shadowColor = 'rgba(255, 0, 110, 0.8)';
    ctx.shadowBlur = 10;
    ctx.beginPath();
    ctx.arc(food.x * gridSize + gridSize/2, food.y * gridSize + gridSize/2, gridSize/2 - 1, 0, Math.PI * 2);
    ctx.fill();
    ctx.shadowColor = 'transparent';
  }
  
  function updateGame() {
    if (!gameRunning) return;
    
    dx = nextDx;
    dy = nextDy;
    
    const head = {x: snake[0].x + dx, y: snake[0].y + dy};
    
    if (head.x < 0 || head.x >= tileCount || head.y < 0 || head.y >= tileCount) {
      gameRunning = false;
      alert('💥 Game Over! Score: ' + score);
      gameReset();
      return;
    }
    
    for (let segment of snake) {
      if (head.x === segment.x && head.y === segment.y) {
        gameRunning = false;
        alert('💥 Game Over! Score: ' + score);
        gameReset();
        return;
      }
    }
    
    snake.unshift(head);
    
    if (head.x === food.x && head.y === food.y) {
      score += 10 + snake.length;
      document.getElementById('score').textContent = score;
      document.getElementById('length').textContent = snake.length;
      food = {x: Math.floor(Math.random() * tileCount), y: Math.floor(Math.random() * tileCount)};
    } else {
      snake.pop();
    }
    
    drawGame();
  }
  
  function gameReset() {
    snake = [{x: 10, y: 10}, {x: 9, y: 10}, {x: 8, y: 10}];
    dx = 1;
    dy = 0;
    nextDx = 1;
    nextDy = 0;
    score = 0;
    gameRunning = true;
    food = {x: 15, y: 15};
    document.getElementById('score').textContent = '0';
    document.getElementById('length').textContent = '3';
    drawGame();
  }
  
  function gamePause() {
    gameRunning = !gameRunning;
  }
  
  document.addEventListener('keydown', (e) => {
    if (e.key === 'ArrowUp' || e.key.toLowerCase() === 'w') { if (dy === 0) { nextDx = 0; nextDy = -1; } }
    if (e.key === 'ArrowDown' || e.key.toLowerCase() === 's') { if (dy === 0) { nextDx = 0; nextDy = 1; } }
    if (e.key === 'ArrowLeft' || e.key.toLowerCase() === 'a') { if (dx === 0) { nextDx = -1; nextDy = 0; } }
    if (e.key === 'ArrowRight' || e.key.toLowerCase() === 'd') { if (dx === 0) { nextDx = 1; nextDy = 0; } }
  });
  
  setInterval(updateGame, 100);
  gameReset();
</script>
